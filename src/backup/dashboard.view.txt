import { NavLink, useNavigate } from "react-router-dom";
import { renderIcon } from "@/utils/icons.utils";
import { useAuth } from "@/hooks/auth/useAuth.hooks";
import Card from "@/components/card.component";
import { getImagePath } from "@/utils/image.utils";

export default function DashboardView({
  recipes,
  count = 0,
  role,
  countAllRecipes = 0,
}) {
  const { me } = useAuth();
  const nav = useNavigate();

  return (
    <>
      <main className="w-full flex flex-col gap-5">
        <div className="p-4 rounded-2xl bg-white grid grid-cols-1 sm:grid-cols-2 md:grid-cols-3 lg:grid-cols-3 gap-2">
          <NavLink
            to={"/dashboard/chef/recipe"}
            className="w-full group relative h-fit bg-blue-700 p-4 hover:scale-110 transition-all duration-300 rounded-2xl text-white flex flex-col"
          >
            <div className="flex items-center justify-between">
              <h1 className="font-bold text-xl">Total Recipes</h1>
              {renderIcon("ArrowUpRight", {
                className:
                  "w-6 h-6 transform group-hover:-translate-y-2 transition-all duration-300 group-hover:translate-x-2",
              })}
            </div>
            <div className="flex items-center gap-2">
              <span className="font-bold text-4xl my-4">
                {countAllRecipes || 0}
              </span>
              <span className="text-xl font-medium">Recipes</span>
            </div>
          </NavLink>
          <NavLink
            className="w-full group relative h-fit bg-amber-700 hover:scale-110 transition-all duration-300 p-4 rounded-2xl text-white flex flex-col"
            to={{
              pathname:
                me?.role === "admin"
                  ? "/dashboard/admin/recipe"
                  : "/dashboard/chef/recipe",
              search: "?status=pending",
            }}
          >
            <div className="flex items-center justify-between">
              <h1 className="font-bold text-xl">Recipe Pending</h1>
              {renderIcon("ArrowUpRight", {
                className:
                  "w-6 h-6 transform group-hover:-translate-y-2 transition-all duration-300 group-hover:translate-x-2",
              })}
            </div>
            <div className="flex items-center gap-2">
              <span className="font-bold text-4xl my-4">
                {count.pending || 0}
              </span>
              <span className="text-xl font-medium">Pending</span>
            </div>
          </NavLink>
          <NavLink
            className="w-full group relative h-fit bg-orange-600 hover:scale-110 transition-all duration-300 p-4 rounded-2xl text-white flex flex-col"
            to={{
              pathname:
                me?.role === "admin"
                  ? "/dashboard/admin/recipe"
                  : "/dashboard/chef/recipe",
              search: "?status=draft",
            }}
          >
            <div className="flex items-center justify-between">
              <h1 className="font-bold text-xl">Recipe Draft</h1>
              {renderIcon("ArrowUpRight", {
                className:
                  "w-6 h-6 transform group-hover:-translate-y-2 transition-all duration-300 group-hover:translate-x-2",
              })}
            </div>
            <div className="flex items-center gap-2">
              <span className="font-bold text-4xl my-4">
                {count.draft || 0}
              </span>
              <span className="text-xl font-medium">Draft</span>
            </div>
          </NavLink>
          <NavLink
            className="w-full group relative h-fit bg-green-700 hover:scale-110 transition-all duration-300 p-4 rounded-2xl text-white flex flex-col"
            to={{
              pathname:
                me?.role === "admin"
                  ? "/dashboard/admin/recipe"
                  : "/dashboard/chef/recipe",
              search: "?status=accept",
            }}
          >
            <div className="flex items-center justify-between">
              <h1 className="font-bold text-xl">Recipe Approved</h1>
              {renderIcon("ArrowUpRight", {
                className:
                  "w-6 h-6 transform group-hover:-translate-y-2 transition-all duration-300 group-hover:translate-x-2",
              })}
            </div>
            <div className="flex items-center gap-2">
              <span className="font-bold text-4xl my-4">
                {count.accept || 0}
              </span>
              <span className="text-xl font-medium">Approved</span>
            </div>
          </NavLink>
          <NavLink
            className="w-full group relative h-fit bg-orange-700 p-4 hover:scale-110 transition-all duration-300 rounded-2xl text-white flex flex-col"
            to={{
              pathname:
                me?.role === "admin"
                  ? "/dashboard/admin/recipe"
                  : "/dashboard/chef/recipe",
              search: "?status=reject",
            }}
          >
            <div className="flex items-center justify-between">
              <h1 className="font-bold text-xl">Recipe Rejected</h1>
              {renderIcon("ArrowUpRight", {
                className:
                  "w-6 h-6 transform group-hover:-translate-y-2 transition-all duration-300 group-hover:translate-x-2",
              })}
            </div>
            <div className="flex items-center gap-2">
              <span className="font-bold text-4xl my-4">
                {count.reject || 0}
              </span>
              <span className="text-xl font-medium">Rejected</span>
            </div>
          </NavLink>
        </div>
        <div className="w-full flex flex-col p-2 lg:p-4 rounded-2xl bg-white">
          <div className="flex mb-5 items-center justify-between p-2 lg:p-0 md:p-0">
            <h1 className="text-xl font-bold">All Recipe</h1>
            <div className="flex items-center text-md text-slate-600 justify-center gap-2">
              <span>Seemore</span>
              {renderIcon("ArrowRight", { className: "w-5 h-5" })}
            </div>
          </div>
          <div className="columns-2 w-full px-2 md:px-0 lg:px-0 space-y-5 lg:space-y-10 md:space-y-0 lg:columns-3">
            {Array.isArray(recipes) && recipes.length > 0 ? (
              recipes.map((item) => {
                return (
                  <Card
                    key={item.id}
                    title={item.title}
                    description={item.description}
                    image={
                      item.image ||
                      getImagePath(`food/item.image`) ||
                      getImagePath("food/food1.jpg")
                    }
                    profile={item.recipes_authors?.profile}
                    author={item.recipes_authors?.username}
                    difficulty={item.difficulty}
                    time={item.time}
                    category={item.category.name}
                    onCardClick={() => nav(`/recipes/detail/${item.id}`)}
                    badge={
                      item.status === "pending"
                        ? "Pending"
                        : item.status === "accept"
                          ? "Disetujui"
                          : item.status === "reject"
                            ? "Ditolak"
                            : "Draft"
                    }
                  />
                );
              })
            ) : (
              <h1 className="text-2xl font-bold text-black">
                Data tidak ditemukan
              </h1>
            )}
          </div>
        </div>
      </main>
    </>
  );
}
